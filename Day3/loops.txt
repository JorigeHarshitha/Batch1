for-loop
syntax:
for(initilization; condition; increment/decrement)
{
    statement
}
eg:-
for(let i=0; i<=5; i++)
{
    console.log(i)
}
//0 1 2 3 4 5

while-loop
-->it will iterate the statement until condition becomes false
syntax:
while(condition)
{
    statement
}
eg:-
let n = 1;
while(n<=5)
{
    console.log(n)
    n++
}

do-while-loop
-->it iterates set of instructions then it will checks condition
syntax:
do{
    statement
}while(condition)
eg:-
let m = 1
do{
    console.log(m)
    m++
}while(m <=5)
